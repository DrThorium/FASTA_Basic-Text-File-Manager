195 struct HuffmanNode *buildHuffman(char codes[], int freqs[], int size){
196 struct HuffmanNode *left, *right, *top;
197 struct MaxHeap *maxHeap = createMaxHeap(codes, freqs, size);
198 while(!isSizeOne(maxHeap)){
199 left = extractMax(maxHeap);
200 right = extractMax(maxHeap);
201 top = newNode('$', left->freq_ + right->freq_);
202 top->left = left;
203 top->right = right;
204 insertMaxHeap(maxHeap, top);
205 }
206 return extractMax(maxHeap);
207 }

180 void insArch(int pos_y, int pos_x, std::vector<int> &Neighs, int char_value) {
181 std::vector<int> vec_(5, -1);
182 if (!checkBase(char_value)){
183 matrix_[pos_x][pos_y] = vec_;
184 }
185 vec_[0] = char_value;
186 for (int i = 0; i < 4; i++) {
187 int exit_w;
188 if (!Sequence::checkBase(Neighs[i])) Neighs[i] = -1;
189 if (Neighs[i] == -1) {
190 exit_w = -1;
191 } else {
192 int Nh = Neighs[i];
193 float diference_ = float(char_value) - float(Nh);
194 if (diference_ == 0) diference_ =+ (float(char_value)/2);
195 diference_ = std::abs(diference_);
196 float weight;
197 weight = 1 / (diference_);
198 weight = weight + 1;
199 exit_w = int(weight);
200 exit_w = std::rand() % 100 + 1;
201 }
202 vec_[i + 1] = exit_w;
203 }
204 matrix_[pos_x][pos_y] = vec_;
205 }
209 void makeGraph(){
210 y_matrix_size_ = lines_list_.size();
211 x_matrix_size_ = max_len_line_;
212 insVertex(y_matrix_size_, x_matrix_size_, matrix_);
213 int pos_y = 0;
214 for (auto str_it = lines_list_.begin(); pos_y < lines_list_.size(); str_it++, pos_y++) {
215 int pos_x = 0;
216 std::string actual, last, next;
217 std::vector<int> vec_NH(4, -1);
218 actual = *str_it;
219 str_it++;
220 if (str_it != lines_list_.end()) {
221 next = *str_it;
222 str_it--;
223 }
224 if (str_it != lines_list_.begin()) {
225 str_it--;
226 last = *str_it;
227 str_it++;
228 }
229 for (; pos_x < actual.size(); pos_x++) {
230 unsigned char actual_char = actual[pos_x];
231 if (pos_x < actual.size()) {
232 unsigned char next_char = actual[pos_x + 1];
233 vec_NH[1] = int(next_char);
234 }
235 if (pos_x != 0) {
236 unsigned char last_char = actual[pos_x - 1];
237 vec_NH[3] = int(last_char);
238 }
239 if (!last.empty()) {
240 if (pos_x < last.size()) {
241 unsigned char up_char = last[pos_x];
242 vec_NH[0] = int(up_char);
243 }
244 }
245 if (!next.empty()) {
246 if (pos_x < next.size()) {
247 unsigned char down_char = next[pos_x];
248 vec_NH[2] = int(down_char);
249 }
250 }
251 insArch(pos_y, pos_x, vec_NH, int(actual_char));

375 void checkNN(int pos_x, int pos_y) {
376 int acumulated_ = tile_matrix_[pos_x][pos_y][0];
377 int up_s, rg_s, lf_s, dw_s;
378 if (tile_matrix_[pos_x][pos_y][2] != -1) {
379 up_s = acumulated_ + tile_matrix_[pos_x][pos_y][2];
380 if (up_s < tile_matrix_[pos_x][pos_y - 1][0]) {
381 tile_matrix_[pos_x][pos_y - 1][0] = up_s;
382 tile_matrix_[pos_x][pos_y - 1][1] = 2;
383 }
384 }
385 if (tile_matrix_[pos_x][pos_y][3] != -1) {
386 rg_s = acumulated_ + tile_matrix_[pos_x][pos_y][3];
387 if (rg_s < tile_matrix_[pos_x + 1][pos_y][0]) {
388 tile_matrix_[pos_x + 1][pos_y][0] = rg_s;
389 tile_matrix_[pos_x + 1][pos_y][1] = 3;
390 }
391 }
392 if (tile_matrix_[pos_x][pos_y][4] != -1) {
393 dw_s = acumulated_ + tile_matrix_[pos_x][pos_y][4];
394 if (dw_s < tile_matrix_[pos_x][pos_y + 1][0]) {
395 tile_matrix_[pos_x][pos_y + 1][0] = dw_s;
396 tile_matrix_[pos_x][pos_y + 1][1] = 0;
397 }
398 }
399 if (tile_matrix_[pos_x][pos_y][5] != -1) {
400 lf_s = acumulated_ + tile_matrix_[pos_x][pos_y][5];
401 if (lf_s < tile_matrix_[pos_x - 1][pos_y][0]) {
402 tile_matrix_[pos_x - 1][pos_y][0] = lf_s;
